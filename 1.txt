local dragging, dragInput, dragStart, startPos
frame.InputBegan:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
        dragging = true; dragStart = input.Position; startPos = frame.Position
        input.Changed:Connect(function() if input.UserInputState == Enum.UserInputState.End then dragging = false end end)
    end
end)
frame.InputChanged:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch then dragInput = input end
end)
UserInputService.InputChanged:Connect(function(input)
    if input == dragInput and dragging then
        local delta = input.Position - dragStart
        frame.Position = UDim2.new(startPos.X.Scale, startPos.X.Offset + delta.X, startPos.Y.Scale, startPos.Y.Offset + delta.Y)
    end
end)

local function switchTab(tabName)
    if screenGui:FindFirstChild("SettingsOverlay") then screenGui.SettingsOverlay.Visible = false end
    for name, tabFrame in pairs(tabs) do
        tabFrame.Visible = (name == tabName)
    end
    for name, btnData in pairs(tabButtons) do
        local isSel = (name == tabName)
        btnData.btn.BackgroundTransparency = isSel and 0 or 1
        btnData.label.TextColor3 = isSel and Theme.Text or Theme.Muted
        btnData.indicator.Visible = isSel
        btnData.indicator.BackgroundColor3 = Theme.Accent
        if btnData.iconLabel then
            btnData.iconLabel.TextColor3 = (isSel and not btnData.noAccentIcon) and Theme.Accent or Theme.Muted
        end
    end
end

local function performTpAttack(targetHrp)
    if not character or not character:FindFirstChild("HumanoidRootPart") then return end
    isAttacking = true
    local myHrp = character.HumanoidRootPart
    local originCFrame = myHrp.CFrame
    myHrp.CFrame = targetHrp.CFrame * CFrame.new(0, 0, 3) 
    myHrp.CFrame = CFrame.new(myHrp.Position, targetHrp.Position) 
    myHrp.Velocity = Vector3.new(0,0,0)
    RunService.RenderStepped:Wait() 
    
    local isFocused = true
    pcall(function() isFocused = type(iswindowactive) == "function" and iswindowactive() or UserInputService:GetWindowFocused() end)
    if isFocused and type(mouse1click) == "function" then pcall(mouse1click) else
        pcall(function() VirtualInputManager:SendMouseButtonEvent(0, 0, 0, true, game, 1); task.wait(0.02); VirtualInputManager:SendMouseButtonEvent(0, 0, 0, false, game, 1) end)
    end
    
    local tool = character:FindFirstChildOfClass("Tool")
    if tool then tool:Activate() end
    RunService.RenderStepped:Wait() 
    if character:FindFirstChild("HumanoidRootPart") then
        character.HumanoidRootPart.CFrame = originCFrame
        character.HumanoidRootPart.Velocity = Vector3.new(0,0,0)
    end
    task.wait(0.05)
    isAttacking = false
end

local function AuraLoop()
    if killAuraEnabled and character then
        local myHrp = character:FindFirstChild("HumanoidRootPart")
        if myHrp then
            if killAuraVisualPart then
                killAuraVisualPart.Size = Vector3.new(killAuraRadius * 2, killAuraRadius * 2, killAuraRadius * 2)
                if not isAttacking then killAuraVisualPart.CFrame = myHrp.CFrame end
            end
            if isAttacking then return end
            
            local overlapParams = OverlapParams.new()
            overlapParams.FilterType = Enum.RaycastFilterType.Exclude
            local ignoreList = {character}
            if killAuraVisualPart then table.insert(ignoreList, killAuraVisualPart) end
            overlapParams.FilterDescendantsInstances = ignoreList
            
            local partsInRadius = workspace:GetPartBoundsInRadius(myHrp.Position, killAuraRadius, overlapParams)
            local closestHrp, closestDist = nil, killAuraRadius + 1
            local processedModels = {}
            for _, part in ipairs(partsInRadius) do
                local model = part:FindFirstAncestorOfClass("Model")
                if model and not processedModels[model] then
                    processedModels[model] = true
                    local eHum = model:FindFirstChildOfClass("Humanoid")
                    local eHrp = model:FindFirstChild("HumanoidRootPart")
                    if eHum and eHrp and eHum.Health > 0 then
                        local dist = (myHrp.Position - eHrp.Position).Magnitude
                        if dist < closestDist then closestDist = dist; closestHrp = eHrp end
                    end
                end
            end
            if closestHrp then task.spawn(performTpAttack, closestHrp) end
        end
    end
end

        if not customDamageConnection then

                local containers = {character, player:FindFirstChild("Backpack")}
                for _, container in ipairs(containers) do
                    if container then
                        for _, v in pairs(container:GetDescendants()) do
                            if (v:IsA("NumberValue") or v:IsA("IntValue")) and string.find(v.Name:lower(), "damage") then v.Value = customDamageValue end
                            if v:IsA("Tool") or v:IsA("Part") then
                                for attrName, attrVal in pairs(v:GetAttributes()) do
                                    if type(attrVal) == "number" and string.find(attrName:lower(), "damage") then v:SetAttribute(attrName, customDamageValue) end
                                end
                            end
                        end
                    end
                end
            end)
        end
    else if customDamageConnection then customDamageConnection:Disconnect(); customDamageConnection = nil end end

    end
